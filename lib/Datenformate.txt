/* Spotify - Objects (abridged for beatdrinks) */
body = {
    seeds: Array[RecommendationSeedObject],
    tracks: Array[TrackObject]
}

seed = {
    initialPoolSize: Number,
    id: String,                 //'disney' ...
    type: String                //'GENRE' / 'TRACK' / 'ARTIST'
}

track = {
    artists: Array[ArtistObject],
    external_ids: ExternalIdObject,     //may contain isrc: String
    external_urls: ExternalUrlObject,   //may contain spotify: String
    href: String,                       //further dialog ( v1/tracks/{id} )
    name: String
}

artistObject = {
    name: String,
    type: 'ARTIST',
    id: String,
    uri: String     //Spotify URI for artist
    external_urls: ExternalUrlObject,
    href: String
}

beatdrinksTracks = Array[beatdrinksTrack]

beatdrinksTrack = {
    id: Number {
        songname: String,
        isrc: String?,
        URL_endUser: String?,
        href: String,           //further dialog ( v1/tracks/{id} )
        artists: Array[ArtistObject]
    }
}
